# Using with CSS-in-JS

## with emotion

```tsx
import { createStyled, injectEmotionStyles } from '@toss-ui/styled';

const styled = createStyled({ injectStyles: injectEmotionStyles });

function Button(props) {
  return <button {...props} />;
}

export const EmotionButton = styled(Button, {
  backgroundColor: 'blueviolet',
  border: 0,
  borderRadius: 12,
  color: 'white',
  fontWeight: 'bold',
  padding: '12px 16px',
});
```

</Canvas>
