# Responsive Styles

## with emotion

```tsx
const styled = createStyled({
  injectStylesGenerator: /* emotion, stitches, ... */,
  media: {
    sm: '(min-width: 640px)',
    md: '(min-width: 768px)',
    lg: '(min-width: 1024px)',
  },
});

export const ResponsiveButton = styled(Button, {
  border: 0,
  borderRadius: 12,
  fontWeight: 'bold',
  padding: '12px 16px',

  backgroundColor: 'blueviolet',
  color: 'white',
  '@sm': {
    backgroundColor: 'red',
  },
  '@md': {
    backgroundColor: 'yellow',
    color: 'black',
  },
  '@lg': {
    backgroundColor: 'green',
  },
});
```
